# Section 06: Report Generation Pipeline

## Overview

The report generation pipeline produces scheduled reports from aggregated data. Reports are generated by querying data sources, rendering visualizations, and converting to multiple output formats (PDF, HTML, CSV). Reports are stored for historical reference and delivered via digest or on-demand.

## Architecture

## Design Decisions

- **Pipeline Architecture**: Separate query, render, and convert stages
- **Multi-Format Output**: Generate all formats from single pipeline run
- **Template-Based**: Handlebars templates for customizable report layouts
- **Caching**: Frequently used queries cached with configurable TTL

## Implementation Approach

```typescript
interface ReportConfig {
  id: string;
  tenantId: string;
  name: string;
  type: 'agent_performance' | 'call_analytics' | 'usage' | 'compliance';
  schedule?: string; // cron expression
  formats: ('pdf' | 'html' | 'csv')[];
  sections: ReportSection[];
  dataSource: {
    type: 'sql' | 'prometheus' | 'clickhouse';
    query: string;
    parameters?: Record<string, unknown>;
  };
}

interface ReportSection {
  id: string;
  title: string;
  type: 'table' | 'chart' | 'summary';
  config: {
    columns?: string[];
    chartType?: 'line' | 'bar' | 'pie';
    metric?: string;
  };
}

interface Report {
  id: string;
  configId: string;
  tenantId: string;
  generatedAt: Date;
  period: { start: Date; end: Date };
  files: Array<{ format: string; url: string; size: number }>;
  metadata: {
    queryDurationMs: number;
    renderDurationMs: number;
    rowCount: number;
  };
}

class ReportGenerator {
  async generate(config: ReportConfig): Promise<Report> {
    const startTime = Date.now();

    // 1. Fetch data
    const queryStart = Date.now();
    const rawData = await this.fetchData(config.dataSource);
    const queryDuration = Date.now() - queryStart;

    // 2. Generate HTML report
    const renderStart = Date.now();
    const html = await this.renderReport(config, rawData);
    const renderDuration = Date.now() - renderStart;

    // 3. Convert to requested formats
    const files: Report['files'] = [];

    for (const format of config.formats) {
      const result = await this.convertFormat(html, format, config);
      files.push(result);
    }

    return {
      id: crypto.randomUUID(),
      configId: config.id,
      tenantId: config.tenantId,
      generatedAt: new Date(),
      period: this.computePeriod(config.schedule),
      files,
      metadata: {
        queryDurationMs: queryDuration,
        renderDurationMs: renderDuration,
        rowCount: rawData.length,
      },
    };
  }

  private async fetchData(source: ReportConfig['dataSource']): Promise<Record<string, unknown>[]> {
    switch (source.type) {
      case 'sql':
        return this.queryService.query(source.query, source.parameters);
      case 'prometheus':
        return this.prometheusService.query(source.query);
      case 'clickhouse':
        return this.clickhouseService.query(source.query);
      default:
        throw new Error(`Unknown data source: ${source.type}`);
    }
  }

  private async renderReport(
    config: ReportConfig,
    data: Record<string, unknown>[],
  ): Promise<string> {
    const template = await this.loadTemplate(config.type);
    const sections = await Promise.all(
      config.sections.map(section => this.renderSection(section, data))
    );

    const rendered = template({
      reportName: config.name,
      generatedAt: new Date().toISOString(),
      period: this.computePeriod(config.schedule),
      sections,
    });

    return rendered;
  }

  private async renderSection(
    section: ReportSection,
    data: Record<string, unknown>[],
  ): Promise<Record<string, unknown>> {
    switch (section.type) {
      case 'table':
        return this.renderTableSection(section, data);
      case 'chart':
        return this.renderChartSection(section, data);
      case 'summary':
        return this.renderSummarySection(section, data);
      default:
        return { title: section.title, type: 'empty', content: '' };
    }
  }

  private async renderTableSection(
    section: ReportSection,
    data: Record<string, unknown>[],
  ): Promise<Record<string, unknown>> {
    const columns = section.config.columns || Object.keys(data[0] || {});
    const rows = data.map(row => columns.map(col => row[col]));

    return {
      type: 'table',
      title: section.title,
      columns,
      rows,
    };
  }

  private async renderChartSection(
    section: ReportSection,
    data: Record<string, unknown>[],
  ): Promise<Record<string, unknown>> {
    // Generate chart as base64 PNG using a chart rendering service
    const chartConfig = {
      type: section.config.chartType || 'line',
      data,
      metric: section.config.metric,
    };

    const chartImage = await this.chartRenderer.render(chartConfig);

    return {
      type: 'chart',
      title: section.title,
      chart: chartImage, // base64 data URL
    };
  }

  private async convertFormat(
    html: string,
    format: string,
    config: ReportConfig,
  ): Promise<{ format: string; url: string; size: number }> {
    switch (format) {
      case 'pdf': {
        const pdf = await this.htmlToPdf(html);
        const url = await this.uploadFile(`${config.id}.pdf`, pdf);
        return { format: 'pdf', url, size: pdf.length };
      }
      case 'html': {
        const url = await this.uploadFile(`${config.id}.html`, Buffer.from(html));
        return { format: 'html', url, size: html.length };
      }
      case 'csv': {
        const csv = this.convertToCsv(html);
        const url = await this.uploadFile(`${config.id}.csv`, Buffer.from(csv));
        return { format: 'csv', url, size: csv.length };
      }
      default:
        throw new Error(`Unsupported format: ${format}`);
    }
  }

  private async htmlToPdf(html: string): Promise<Buffer> {
    const browser = await puppeteer.launch();
    const page = await browser.newPage();
    await page.setContent(html, { waitUntil: 'networkidle0' });
    const pdf = await page.pdf({ format: 'A4', printBackground: true });
    await browser.close();
    return Buffer.from(pdf);
  }

  private convertToCsv(html: string): string {
    // Simple CSV extraction from HTML tables
    const tableRegex = /<table>([\s\S]*?)<\/table>/g;
    const rows: string[] = [];
    let match;

    while ((match = tableRegex.exec(html)) !== null) {
      const tableHtml = match[1];
      const csvRows = tableHtml.match(/<tr>([\s\S]*?)<\/tr>/g) || [];
      for (const row of csvRows) {
        const cells = row.match(/<t[dh]>([\s\S]*?)<\/t[dh]>/g) || [];
        const csvRow = cells.map(c => {
          const val = c.replace(/<[^>]*>/g, '').trim();
          return val.includes(',') ? `"${val}"` : val;
        }).join(',');
        rows.push(csvRow);
      }
    }

    return rows.join('\n');
  }
}
```

## Integration Points

- **Analytics DB**: PostgreSQL/ClickHouse for report data queries
- **Chart Rendering**: Headless chart rendering service
- **S3/Cloud Storage**: Report file storage
- **Digest Engine**: Reports included as digest content modules

## Production Considerations

- **Report Size Limits**: Maximum 20MB per report file
- **Query Timeouts**: 5-minute query timeout for report data
- **Scheduled Generation**: Reports generated during off-peak hours
- **Cost Tracking**: Report generation costs tracked per tenant

## Open-Source Tools

- **Handlebars**: HTML template engine
- **Puppeteer**: HTML to PDF conversion
- **PapaParse**: CSV parsing and generation
- **Chart.js**: Chart rendering in HTML reports
