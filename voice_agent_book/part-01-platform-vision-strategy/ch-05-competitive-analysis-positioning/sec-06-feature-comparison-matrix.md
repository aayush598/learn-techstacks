## Feature Comparison Data Model

```typescript
interface FeatureComparison {
  category: string;
  features: FeatureRow[];
}

interface FeatureRow {
  name: string;
  importance: 'critical' | 'important' | 'nice-to-have' | 'future';
  ourScore: 0 | 1 | 2 | 3; // 0=missing, 1=partial, 2=parity, 3=superior
  competitors: Record<string, 0 | 1 | 2 | 3>;
}

function calculateCompetitiveScore(
  matrix: FeatureComparison[],
  targetSegment: string
): CompetitiveScore {
  let totalWeight = 0;
  let ourScore = 0;
  const competitorScores: Record<string, number> = {};
  
  for (const category of matrix) {
    for (const feature of category.features) {
      const weight = feature.importance === 'critical' ? 3 : 
                     feature.importance === 'important' ? 2 : 1;
      totalWeight += weight;
      ourScore += feature.ourScore * weight;
      
      for (const [competitor, score] of Object.entries(feature.competitors)) {
        competitorScores[competitor] = (competitorScores[competitor] || 0) + (score * weight);
      }
    }
  }
  
  return {
    ourScore: ourScore / totalWeight,
    competitorScores: Object.fromEntries(
      Object.entries(competitorScores).map(([k, v]) => [k, v / totalWeight])
    ),
    strongestAreas: findStrengths(matrix),
    weakestAreas: findWeaknesses(matrix),
  };
}
```
